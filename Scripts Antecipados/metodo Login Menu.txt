static Cliente LogarCliente(Cliente[] clientes, int qtdClientes)
{
    Console.Write("Login: ");
    string login = Console.ReadLine();

    Console.Write("Senha: ");
    string senha = Console.ReadLine();

    for (int i = 0; i < qtdClientes; i++)
    {
        if (clientes[i].Login == login && clientes[i].Senha == senha)
        {
            clientes[i].Logado = true;
            return clientes[i];
        }
    }

    return null;
}
